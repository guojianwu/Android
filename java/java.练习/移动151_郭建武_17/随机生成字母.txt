
public class guo{
	public static void main(String[] args){
		char c= (char)(Math.random()*26+'a');
		System.out.println(c+":");
		switch(c){
		case 'a':
		case 'e':	
		case 'i':	
		case 'o':	
		case 'u':	
			System.out.println("是元音");
		    break;
		case 'y':	
		case 'w':	 
			System.out.println("是辅音");
			break;
		default:	
			System.out.println("普通字母");
		}
	
	}

}
