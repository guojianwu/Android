import java.awt.Color;
import java.awt.Frame;
import java.awt.TextArea;
import java.awt.event.AdjustmentEvent;
import java.awt.event.AdjustmentListener;
import java.awt.event.WindowAdapter;
import java.awt.event.WindowEvent;

import javax.swing.Box;
import javax.swing.JScrollBar;



public class GUI_2 {

	Frame f = new Frame("调整事件事例");
	
	TextArea ta = new TextArea(10,30);
	 JScrollBar red=new JScrollBar(JScrollBar.HORIZONTAL,0,1,0,255);
	 JScrollBar green=new JScrollBar(JScrollBar.HORIZONTAL,0,1,0,255);
	 JScrollBar blue=new JScrollBar(JScrollBar.HORIZONTAL,0,1,0,255);
	 int red1=0,green1=0,blue1=0;
	 public void GUI_2(){
		 Box box = Box.createVerticalBox();
		 box.add(ta);
		 box.add(red);
		 box.add(green);
		 box.add(blue);
		 f.add(box);
		 
		 ta.setBackground(new Color(red1,green1,blue1));
		 AdjustmentListener a = new AdjustmentListener(){
			public void adjustmentValueChanged(AdjustmentEvent e) {
				if(e.getSource()==red)
					red1=e.getValue();
				if(e.getSource()==green)
					green1=e.getValue();
				if(e.getSource()==blue)
					blue1=e.getValue();
				ta.setBackground(new Color(red1,green1,blue1));
			}	 
		 };
		 red.addAdjustmentListener(a);
		 green.addAdjustmentListener(a);
		 blue.addAdjustmentListener(a);
		 WindowAdapter w = new WindowAdapter(){

			@Override
			public void windowClosing(WindowEvent e) {
				System.exit(0);
			}
			 
		 };
		 f.addWindowListener(w);
		 f.setVisible(true);
		 f.pack();
	 }
	 public static void main(String[] args) {
		new GUI_2().GUI_2();
	}
}
