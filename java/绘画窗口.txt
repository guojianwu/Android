import java.awt.BorderLayout;
import java.awt.Button;
import java.awt.Canvas;
import java.awt.Color;
import java.awt.Dimension;
import java.awt.Frame;
import java.awt.Graphics;
import java.awt.Panel;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.util.Random;





public class  a{
	Frame f=new Frame("简单绘图");
	Button rect=new Button("绘制矩形");
	Button oval=new Button("绘制圆形");
	MyCanvas drawArea =new MyCanvas();
	String shape="rect";  //shape保存要绘制的形状标识，只能取"rect"和"oval"
	
	
	public void init(){
		Panel p=new Panel();
		p.add(rect);
		p.add(oval);
		
		drawArea.setPreferredSize(new Dimension(250,180)); //设置画布的大小
		f.add(drawArea);
		f.add(p,BorderLayout.SOUTH);
		
		ActionListener l=new ActionListener() {
			public void actionPerformed(ActionEvent e) {
				//if(e.getActionCommand().equals("绘制矩形"));
				if(e.getSource().equals(rect)){
					//绘制空心矩形
					shape="rect";
					drawArea.repaint();
					
				}
				if(e.getSource().equals(oval)){
					//绘制实心圆形
					shape="oval";
					drawArea.repaint();
				}
				
			}
		};
		
		rect.addActionListener(l);
		oval.addActionListener(l);
		
		f.pack();
		f.setVisible(true);
	}
	
	class MyCanvas extends Canvas{
		public void paint(Graphics g) {
		     //通过paint方法绘制形状
			if(shape.equals("rect")){
				g.setColor(new Color(220,100,80));
				Random rand=new Random();
				
				g.drawRect(rand.nextInt(200), rand.nextInt(200), 40, 60);
			}
			if(shape.equals("oval")){
				g.setColor(new Color(80,100,200));
				Random rand=new Random();
				g.fillOval(rand.nextInt(200), rand.nextInt(120), 50, 50);
				
			}
		}
		}
		public static void main(String[] args) {
			new a().init();
		}
		
	}
